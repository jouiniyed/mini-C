void rec_comparison(AstNode **A){

    AstNode *B;
    AstNode *C;
    TypeComparisonOperator op;
    rec_gae(&B);
    if (current_token().type == EQUAL ||
    current_token().type == NOT_EQUAL ||
    current_token().type == LESS ||
    current_token().type == GREATER ||
    current_token().type == LESS_EQUAL ||
    current_token().type == GREATER_EQUAL) {
        op = get_comparison_operator(current_token().type);
        advance();

        if(current_token().type == END_OF_SEQUENCE || current_token().type == SEMI){
            printf("ERROR: Missing right-hand expression after comparison\n");
            exit(1);
        }

        rec_comparison(&C);
        *A = create_comparison_operation(op,B,C);
    } else {
        *A = B;
    }


}
